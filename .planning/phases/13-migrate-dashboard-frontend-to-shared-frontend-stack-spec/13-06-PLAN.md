       ```dockerfile
       # ── bun builder ──────────────────────────────────────────────
       FROM oven/bun:1 AS dashboard-build
       WORKDIR /build/dashboard
       COPY dashboard/package.json dashboard/bun.lockb ./
       RUN bun install --frozen-lockfile
       COPY dashboard/ ./
       RUN bun run build

       # ── runtime ──────────────────────────────────────────────────
       FROM python:3.12-slim AS runtime
       # install node + bun runtime (bun standalone binary copied)
       COPY --from=oven/bun:1 /usr/local/bin/bun /usr/local/bin/bun
       RUN apt-get update && apt-get install -y --no-install-recommends curl ca-certificates git \
           && curl -fsSL https://deb.nodesource.com/setup_22.x | bash - \
           && apt-get install -y nodejs \
           && rm -rf /var/lib/apt/lists/*
       # Claude Code CLI (existing step)
       RUN npm i -g @anthropic-ai/claude-code@latest
       WORKDIR /app
       COPY pyproject.toml ./
       RUN pip install --no-cache-dir .
       COPY bot/ ./bot/
       COPY scripts/ ./scripts/
       # Dashboard build artifacts
       COPY --from=dashboard-build /build/dashboard/.next/standalone ./dashboard/
       COPY --from=dashboard-build /build/dashboard/.next/static ./dashboard/.next/static
       COPY --from=dashboard-build /build/dashboard/public ./dashboard/public
       EXPOSE 8090
       CMD ["python", "-m", "bot"]
       ```

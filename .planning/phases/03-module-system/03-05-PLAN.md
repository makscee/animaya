    from pathlib import Path

    pytest.importorskip("bot.modules", reason="bot.modules package not yet implemented")

    from bot.modules import (  # noqa: E402

        get_entry,
        install,
        list_installed,
        uninstall,

    REPO_ROOT = Path(__file__).resolve().parent.parent.parent
    MODULES_ROOT = REPO_ROOT / "modules"


    class TestRoundtrip:
        """Install + uninstall the first-party bridge module end-to-end."""

        def test_install_uninstall_leaves_no_artifacts(self, tmp_hub_dir: Path) -> None:
            # Install
            entry = install("bridge", modules_root=MODULES_ROOT, hub_dir=tmp_hub_dir, config={})
            assert entry["name"] == "bridge"
            assert entry["version"] == "1.0.0"
            assert get_entry(tmp_hub_dir, "bridge") is not None
            assert list_installed(tmp_hub_dir) == ["bridge"]

            # Assembled CLAUDE.md should include the bridge section
            claude_md = tmp_hub_dir / "CLAUDE.md"
            assert claude_md.is_file()
            content = claude_md.read_text(encoding="utf-8")
            assert '<module name="bridge">' in content
            assert "Telegram Bridge" in content  # from prompt.md
            assert "</module>" in content

            # Uninstall
            uninstall("bridge", modules_root=MODULES_ROOT, hub_dir=tmp_hub_dir)
            assert get_entry(tmp_hub_dir, "bridge") is None
            assert list_installed(tmp_hub_dir) == []

            # Assembled CLAUDE.md should be back to empty-modules marker
            content_after = claude_md.read_text(encoding="utf-8")
            assert "<!-- No modules installed -->" in content_after
            assert '<module name="bridge">' not in content_after

        def test_registry_empty_after_roundtrip(self, tmp_hub_dir: Path) -> None:
            install("bridge", modules_root=MODULES_ROOT, hub_dir=tmp_hub_dir)
            uninstall("bridge", modules_root=MODULES_ROOT, hub_dir=tmp_hub_dir)
            assert list_installed(tmp_hub_dir) == []

        def test_owned_paths_empty_means_no_leakage_check_artifacts(
            self, tmp_hub_dir: Path
        ) -> None:
            """Bridge declares owned_paths=[] so no hub-side cleanup is expected."""
            install("bridge", modules_root=MODULES_ROOT, hub_dir=tmp_hub_dir)
            # The install.sh does not create any owned_paths; verify none exist.
            # (List hub_dir before and after; only CLAUDE.md + registry.json should appear.)
            hub_files_before = set(p.name for p in tmp_hub_dir.iterdir())
            uninstall("bridge", modules_root=MODULES_ROOT, hub_dir=tmp_hub_dir)
            hub_files_after = set(p.name for p in tmp_hub_dir.iterdir())
            # No spurious files created or left behind that weren't there before install
            # (registry.json and CLAUDE.md expected in both states after first install)
            allowed = {"CLAUDE.md", "registry.json"}
            assert hub_files_before <= allowed, f"unexpected hub files: {hub_files_before}"
            assert hub_files_after <= allowed, f"unexpected hub files: {hub_files_after}"

    # Animaya Module Authoring Guide

    This guide describes how to author a module for Animaya v2's module system
    (Phase 3, MODS-01..MODS-06). Read it before creating a new module.

    ## Module layout

    Every module is a directory under `~/animaya/modules/<name>/` containing:

    ```
    modules/<name>/
    ├── manifest.json     # Required — pydantic-validated metadata
    ├── install.sh        # Required — lifecycle install hook
    ├── uninstall.sh      # Required — lifecycle uninstall hook
    └── prompt.md         # Required — system-prompt snippet injected into CLAUDE.md
    ```

    The first-party `modules/bridge/` module is a minimal reference.

    ## manifest.json schema (pydantic-strict)

    Required fields:
    - `manifest_version` (int, ≥1): currently 1.
    - `name` (string, `^[a-z][a-z0-9_-]*$`): MUST match the folder name.
    - `version` (string, `^\d+\.\d+\.\d+.*$`): semver prefix (e.g., `1.0.0`, `1.2.3-beta`).
    - `system_prompt_path` (string): relative path to prompt file within the module dir.
    - `owned_paths` (array of strings): every file or directory this module creates
      under the Hub (relative to `ANIMAYA_HUB_DIR`, never absolute, no `..`).

    Optional fields:
    - `scripts.install` (string, default `"install.sh"`).
    - `scripts.uninstall` (string, default `"uninstall.sh"`).
    - `depends` (array of module names, default `[]`).
    - `config_schema` (JSON Schema dict, default `null`): rendered by the Phase 5
      dashboard to collect user config.

    **Strict validation:** unknown fields are REJECTED. Schema evolution via
    `manifest_version` bump.

    ## Lifecycle contract (D-11)

    Both `install.sh` and `uninstall.sh` are invoked as `bash <script>` with:

    - Working directory: the module directory.
    - Env vars injected:
      - `ANIMAYA_MODULE_DIR` — absolute path to this module's directory.
      - `ANIMAYA_HUB_DIR`    — absolute path to `~/hub/knowledge/animaya/`.
      - `ANIMAYA_CONFIG_JSON` — JSON string of user config (may be `{}`).
    - Exit code: `0` on success, non-zero on failure.

    **Install failure triggers auto-rollback** (D-13): the installer runs your
    `uninstall.sh` best-effort to clean partial state, then verifies no
    `owned_paths` remain. No registry entry is written.

    **Reinstall is rejected** (D-14): installing an already-installed module
    raises a clear error. User must uninstall first.

    ## Idempotency requirement (CRITICAL)

    `uninstall.sh` MUST be idempotent and MUST NOT fail when run on partial or
    absent state. The installer may call it mid-rollback after install.sh has
    only partially created owned_paths.

    Good pattern:
    ```bash
    #!/usr/bin/env bash
    set -euo pipefail
    rm -rf "${ANIMAYA_HUB_DIR}/my-module-data"   # rm -rf is idempotent
    ```

    Bad pattern (fails if dir missing):
    ```bash
    rmdir "${ANIMAYA_HUB_DIR}/my-module-data"    # fails if empty-dir missing
    ```

    ## owned_paths rules (MODS-05)

    Every file or directory your module creates under the Hub MUST be listed in
    `owned_paths`. The installer verifies these are absent after uninstall (the
    leakage check) and rejects absolute paths and `..` traversal segments at
    install time.

    - RELATIVE to `ANIMAYA_HUB_DIR`.
    - No leading `/`, no `..`.
    - Directories are OK (their contents are cleaned with them).
    - Example: `["identity/me.md", "identity/owner.md"]`.

    ## Dependencies (D-15)

    `depends` lists other module names that must already be installed before
    this one. No auto-install cascade — the installer rejects the install with
    `missing dependency for <name>: [...]`.

    Uninstall is also blocked if another installed module depends on yours.

    ## prompt.md — system prompt injection (D-17)

    Your module's prompt.md content is wrapped in `<module name="{your_name}">`
    in the assembled CLAUDE.md. Keep prompts focused and concise — a module's
    prompt is injected into every Claude call the bot makes.

    DO NOT include the literal string `</module>` in prompt.md — the assembler
    escapes it as `&lt;/module&gt;` to prevent section-boundary injection, which
    may alter your intended rendering.

    ## MODS-06 — module isolation (CONVENTION)

    Modules MUST NOT import code from sibling modules. Communication happens
    exclusively through shared Hub files (markdown, JSON) under
    `ANIMAYA_HUB_DIR`. If you find yourself reaching across modules, it means
    the shared state belongs in a common Hub file instead.

    This rule is convention-enforced in Phase 3 (no lint rule) and validated
    by the Phase 3 `tests/modules/test_isolation.py` AST check.

    ## Anti-patterns (avoid)

    - **Writing to paths not in `owned_paths`.** The leakage check will miss
      them; uninstall leaves garbage.
    - **Absolute paths in `owned_paths`.** Rejected at install time.
    - **Mutable install side-effects outside the hub.** Install must not modify
      the repo, system config, or user's home beyond the hub dir.
    - **Runtime pip / npm.** Add deps to `pyproject.toml` (Python) or document
      system requirements; the installer does NOT resolve package deps.
    - **Silent failure.** Scripts must `set -euo pipefail` and exit non-zero
      on any error so the installer can roll back.
    - **Storing secrets in `prompt.md`.** prompt.md lands verbatim in the
      assembled CLAUDE.md. Use env vars for secrets.

    ## Testing your module

    Use the `modules/bridge/` reference as a starting template. Validate your
    manifest locally:

    ```bash
    python -c "from bot.modules.manifest import validate_manifest; from pathlib import Path; print(validate_manifest(Path('modules/your-module')))"
    ```

    Dry-run install against an isolated tmp hub:

    ```python
    from bot.modules import install, uninstall
    from pathlib import Path
    tmp_hub = Path('/tmp/animaya-dryrun'); tmp_hub.mkdir(exist_ok=True)
    install("your-module", hub_dir=tmp_hub)
    uninstall("your-module", hub_dir=tmp_hub)
    ```

    ## Reference: bridge module

    See `modules/bridge/` for a zero-`owned_paths` module. Its `install.sh` and
    `uninstall.sh` are near-no-ops because the bridge's Python code lives in
    `bot/bridge/` (managed by git). This exercises the lifecycle plumbing
    without adding hub artifacts.
    ```

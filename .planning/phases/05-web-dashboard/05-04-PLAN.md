      """Bot runtime status probe for the dashboard (Phase 5 D-17).

      Shells out to systemctl --user is-active animaya. Falls back to
      'unknown' in dev environments where systemctl is absent or the unit
      is not installed.
      """
      from __future__ import annotations

      import logging
      import os
      import shutil
      import subprocess
      from datetime import datetime, timedelta, timezone
      from pathlib import Path

      from bot.events import tail as events_tail
      from bot.modules import list_installed

      logger = logging.getLogger(__name__)

      SYSTEMD_UNIT: str = os.environ.get("ANIMAYA_SYSTEMD_UNIT", "animaya")
      SYSTEMD_SCOPE: str = os.environ.get("ANIMAYA_SYSTEMD_SCOPE", "--user")

      STATUS_LABELS: dict[str, tuple[str, str]] = {
          "active":      ("Running", "dot-green"),
          "activating":  ("Starting", "dot-warn"),
          "inactive":    ("Stopped", "dot-red"),
          "failed":      ("Failed", "dot-red"),
          "unknown":     ("Unknown", "dot-gray"),
      }


      def is_running() -> str:
          """Return systemctl is-active state, or 'unknown' on failure."""
          if shutil.which("systemctl") is None:
              return "unknown"
          cmd = ["systemctl", SYSTEMD_SCOPE, "is-active", SYSTEMD_UNIT] if SYSTEMD_SCOPE else ["systemctl", "is-active", SYSTEMD_UNIT]
          try:
              result = subprocess.run(
                  cmd, capture_output=True, text=True, timeout=2, check=False,
              )
          except (subprocess.TimeoutExpired, OSError) as exc:
              logger.debug("systemctl probe failed: %s", exc)
              return "unknown"
          out = (result.stdout or "").strip()
          return out if out in STATUS_LABELS else "unknown"


      def _events_today_count() -> int:
          """Count events emitted within the last 24 hours."""
          cutoff = datetime.now(timezone.utc) - timedelta(hours=24)
          count = 0
          for rec in events_tail(10_000):
              ts = rec.get("ts")
              if not isinstance(ts, str):
                  continue
              try:
                  parsed = datetime.fromisoformat(ts)
              except ValueError:
                  continue
              if parsed >= cutoff:
                  count += 1
          return count


      def recent_stats(hub_dir: Path) -> dict:
          """Bundle status + counts for the status strip fragment."""
          status = is_running()
          label, dot = STATUS_LABELS[status]
          return {
              "status": status,
              "status_label": label,
              "status_dot": dot,
              "modules_installed": len(list_installed(Path(hub_dir))),
              "events_today": _events_today_count(),
          }


      __all__ = ["is_running", "recent_stats", "STATUS_LABELS"]

      """Home page and its HTMX fragment endpoints (Phase 5 DASH-03).

      Registered by bot.dashboard.app.build_app via
      `_register_home_routes_if_available`. Removes the placeholder `/`
      route that build_app installs and replaces it with the real home view.
      """
      from __future__ import annotations

      import logging
      from datetime import datetime, timezone
      from pathlib import Path

      from fastapi import Depends, FastAPI, Request
      from fastapi.responses import HTMLResponse
      from fastapi.templating import Jinja2Templates

      from bot.dashboard.deps import require_owner
      from bot.dashboard.status import recent_stats
      from bot.events import tail as events_tail

      logger = logging.getLogger(__name__)

      _RECENT_LIMIT = 50


      def register(app: FastAPI, templates: Jinja2Templates) -> None:
          """Install home + fragment routes; idempotent."""
          _ensure_relative_time_filter(templates)
          _remove_route(app, name="home_placeholder")

          hub_dir = app.state.hub_dir

          @app.get("/", response_class=HTMLResponse, name="home")
          async def home(request: Request, _uid: int = Depends(require_owner)):
              events = events_tail(_RECENT_LIMIT)
              errors = [e for e in events if e.get("level") == "error"]
              return templates.TemplateResponse("home.html", {
                  "request": request,
                  "stats": recent_stats(hub_dir),
                  "events": events,
                  "errors": errors,
                  "nav_active": "home",
              })

          @app.get("/fragments/status", response_class=HTMLResponse, name="fragments_status")
          async def fragments_status(request: Request, _uid: int = Depends(require_owner)):
              return templates.TemplateResponse(
                  "_fragments/status_strip.html",
                  {"request": request, "stats": recent_stats(hub_dir)},
              )

          @app.get("/fragments/activity", response_class=HTMLResponse)
          async def fragments_activity(request: Request, _uid: int = Depends(require_owner)):
              return templates.TemplateResponse(
                  "_fragments/activity_feed.html",
                  {"request": request, "events": events_tail(_RECENT_LIMIT)},
              )

          @app.get("/fragments/errors", response_class=HTMLResponse)
          async def fragments_errors(request: Request, _uid: int = Depends(require_owner)):
              all_ev = events_tail(_RECENT_LIMIT)
              errors = [e for e in all_ev if e.get("level") == "error"]
              return templates.TemplateResponse(
                  "_fragments/error_feed.html",
                  {"request": request, "events": errors},
              )


      def _remove_route(app: FastAPI, *, name: str) -> None:
          app.router.routes = [r for r in app.router.routes if getattr(r, "name", None) != name]


      def _ensure_relative_time_filter(templates: Jinja2Templates) -> None:
          """Add `relative_time` Jinja filter if not already present."""
          env = templates.env
          if "relative_time" in env.filters:
              return
          def _rel(ts: str) -> str:
              try:
                  parsed = datetime.fromisoformat(ts)
              except (TypeError, ValueError):
                  return ""
              now = datetime.now(timezone.utc)
              delta = now - parsed
              s = int(delta.total_seconds())
              if s < 60: return f"{s}s ago"
              if s < 3600: return f"{s // 60}m ago"
              if s < 86400: return f"{s // 3600}h ago"
              return f"{s // 86400}d ago"
          env.filters["relative_time"] = _rel


      __all__ = ["register"]

      - imports: os, logging, time, urllib.parse, pathlib.Path, typing, fastapi (FastAPI, Request, Response, Form, Cookie, HTTPException, Depends, status), fastapi.responses (HTMLResponse, RedirectResponse), fastapi.staticfiles.StaticFiles, fastapi.templating.Jinja2Templates, bot.dashboard.auth (SESSION_COOKIE_NAME, SESSION_MAX_AGE_SECONDS, verify_telegram_payload, issue_session_cookie, read_session_cookie, clear_session_cookie_kwargs), bot.dashboard.deps (require_owner).

          TEMPLATES_DIR = Path(__file__).parent / "templates"
          STATIC_DIR = Path(__file__).parent / "static"

      - def build_app(hub_dir: Path) -> FastAPI:
          app = FastAPI(title="Animaya Dashboard", docs_url=None, redoc_url=None)
          app.state.hub_dir = Path(hub_dir).resolve()
          app.mount("/static", StaticFiles(directory=str(STATIC_DIR)), name="static")
          _register_auth_routes(app)
          _register_home_placeholder(app)
          _register_home_routes_if_available(app)   # Plan 04
          _register_module_routes_if_available(app) # Plan 05 (adds /modules)
          return app

      - def _register_auth_routes(app: FastAPI) -> None:
          @app.get("/login", response_class=HTMLResponse)
          async def login(request: Request, error: str | None = None):
              bot_username = os.environ.get("TELEGRAM_BOT_USERNAME", "").strip()
              return templates.TemplateResponse("login.html", {
                  "request": request,
                  "bot_username": bot_username,
                  "error": error if bot_username else "misconfigured",
              })

          @app.post("/auth/telegram")
          async def auth_telegram(request: Request):
              form = await request.form()
              payload = {k: str(v) for k, v in form.multi_items()}
              bot_token = os.environ.get("TELEGRAM_BOT_TOKEN", "")
              if not verify_telegram_payload(payload, bot_token):
                  # Distinguish stale vs invalid via auth_date freshness check
                  try:
                      age = int(time.time()) - int(payload.get("auth_date", 0))
                  except (TypeError, ValueError):
                      age = 0
                  reason = "stale" if age > 86400 else "invalid"
                  return RedirectResponse(f"/login?error={reason}", status_code=303)
              user_id = int(payload["id"])
              # Owner allowlist check
              from bot.dashboard.deps import _owner_ids  # noqa: PLC0415
              if user_id not in _owner_ids():
                  return RedirectResponse("/login?error=forbidden", status_code=303)
              # Mint cookie
              cookie = issue_session_cookie(
                  user_id=user_id,
                  auth_date=int(payload.get("auth_date", time.time())),
                  hash_=payload.get("hash", ""),
              )
              response = RedirectResponse("/", status_code=303)
              response.set_cookie(
                  key=SESSION_COOKIE_NAME, value=cookie,
                  max_age=SESSION_MAX_AGE_SECONDS,
                  httponly=True, samesite="lax", secure=True, path="/",
              )
              return response

          @app.get("/logout")
          async def logout(response: Response):
              resp = RedirectResponse("/login", status_code=303)
              resp.delete_cookie(**clear_session_cookie_kwargs())
              return resp

      - def _register_home_placeholder(app: FastAPI) -> None:
          @app.get("/", response_class=HTMLResponse, name="home_placeholder")
          async def home(request: Request, user_id: int = Depends(require_owner)):
              return templates.TemplateResponse("_home_placeholder.html", {
                  "request": request, "user_id": user_id, "nav_active": "home",

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser (unauth) -> /login | Anyone can hit /login; widget iframe renders |
| browser (Telegram callback) -> /auth/telegram | Form-encoded POST carrying id/auth_date/hash/etc. |
| browser (authenticated) -> / and other pages | Cookie re-presented; owner allowlist verified |
| browser -> /static/* | Public, read-only assets |
| Caddy proxy -> uvicorn 127.0.0.1 | Proxy headers trusted only from 127.0.0.1 (Plan 07) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-03-01 | S (Spoofing) | forged /auth/telegram payload | mitigate | Delegate to verify_telegram_payload (Plan 02). No code path on /auth/telegram bypasses it. |
| T-05-03-02 | R (Replay) | auth_date replay beyond 24 h | mitigate | verify_telegram_payload rejects; /auth/telegram 302s to /login?error=stale. |
| T-05-03-03 | E (Elevation) | authenticated non-owner | mitigate | /auth/telegram checks _owner_ids() before minting cookie; require_owner re-checks on every page. |
| T-05-03-04 | T (XSS) | stored XSS via login error param | mitigate | Jinja autoescape on by default (Jinja2Templates enables it). Error codes are a finite enum; render-side if/elif restricts to fixed strings. |
| T-05-03-05 | I (Info Disclosure) | /docs /redoc exposing routes | mitigate | FastAPI(docs_url=None, redoc_url=None) disables OpenAPI UI. |
| T-05-03-06 | T (Tampering) | cookie theft via JS | mitigate | httponly=True, samesite="lax", secure=True on every set_cookie call — verified by Test L4 assertion. |
| T-05-03-07 | D (Supply chain) | unpinned HTMX CDN | mitigate | <script src="https://unpkg.com/htmx.org@2.0.3"> — version pinned. SRI integrity hash is a SUMMARY-recorded follow-up. |
| T-05-03-08 | D (Supply chain) | Telegram widget script | transfer | First-party Telegram CDN; required by spec; trust boundary accepted. |
| T-05-03-09 | T (CSRF) | cross-site POST /auth/telegram | mitigate | SameSite=Lax cookie not yet present when /auth/telegram is POSTed (it is the flow that mints the cookie). Hash verification itself acts as the CSRF token — an attacker cannot forge a valid Telegram HMAC without the bot_token. |
| T-05-03-10 | T (CSRF) | cross-site POST /logout | mitigate | Route is GET not POST per spec. SameSite=Lax cookie protects any future state-changing POSTs (Plans 05/06 inherit). |
</threat_model>

    ```python
    def _seed_telegram_bridge_token(data_path: Path) -> None:
        """One-shot seed: write TELEGRAM_BOT_TOKEN env into module config.json
        if bridge is installed and config has no token. After seed (or if env
        already present in config), env is ignored on subsequent boots.
        """
        entry = get_entry(data_path, "telegram-bridge")
        if entry is None:
            return  # bridge not installed — nothing to seed
        module_dir = Path(entry["module_dir"])
        config_path = module_dir / "config.json"
        existing = {}
        if config_path.exists():
            try:
                existing = json.loads(config_path.read_text())
            except json.JSONDecodeError:
                existing = {}
        if existing.get("token"):
            return  # already has token — env ignored
        env_token = os.environ.get("TELEGRAM_BOT_TOKEN", "")
        if not env_token:
            return  # nothing to seed
        existing["token"] = env_token
        # Atomic write (match registry.py pattern)
        tmp = config_path.with_suffix(".json.tmp")
        tmp.write_text(json.dumps(existing, indent=2))
        tmp.replace(config_path)
        logger.warning(
            "TELEGRAM_BOT_TOKEN seeded into %s; env var is deprecated and "
            "will be ignored after config.json has a token.",
            config_path,
        )
    ```

    Important: The helper does NOT auto-install the bridge. If bridge not in
    registry → silent no-op (matches D-8.4: "Never auto-installs the module. Install
    is explicit").

    ```python
        await ctx.stop_event.wait()
        await supervisor.stop_all()

    async def test_bridge_lifecycle_end_to_end(tmp_path: Path):
        """SC#3: Install → Telegram roundtrip → Uninstall → Silence → Reinstall."""
        data_path = tmp_path / "data"
        data_path.mkdir()

        # Stage 1: Install bridge module with test bot token.
        await lifecycle.install(
            module_dir=Path("modules/telegram-bridge"),
            data_path=data_path,
            config={"token": TEST_BOT_TOKEN},

            dashboard_app=None,

        async with get_client() as bundle:

            async with await start_listening(bundle) as listener2:
                await send_to_bot(bundle, "silence-check")
                with pytest.raises(asyncio.TimeoutError):
                    await wait_for_reply(listener2, timeout=10, settle=1.0)

            await lifecycle.install(
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| TELEGRAM_BOT_TOKEN env → modules/telegram-bridge/config.json | One-shot seed crosses env → filesystem boundary. Atomic write prevents partial state. |

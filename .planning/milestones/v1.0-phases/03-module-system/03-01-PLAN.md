    ```python
    """Module manifest schema (Phase 3, MODS-01).

    Strict pydantic v2 model validating modules/<name>/manifest.json.
    Per D-10, unknown fields are rejected; schema evolution uses manifest_version bump.
    """
    from __future__ import annotations

    import json
    import logging
    from pathlib import Path

    from pydantic import BaseModel, ConfigDict, Field

    logger = logging.getLogger(__name__)

    # ── Schema ──────────────────────────────────────────────────────────
    SEMVER_PREFIX_PATTERN = r"^\d+\.\d+\.\d+.*$"  # loose: major.minor.patch prefix (RESEARCH Open Q 3)

    class ModuleScripts(BaseModel):
        """Lifecycle script filenames (relative to module dir)."""
        model_config = ConfigDict(extra="forbid")
        install: str = "install.sh"
        uninstall: str = "uninstall.sh"


    class ModuleManifest(BaseModel):
        """Strict pydantic model for modules/<name>/manifest.json (D-08, D-09, D-10)."""
        model_config = ConfigDict(extra="forbid")

        manifest_version: int = Field(..., ge=1, description="Schema version; currently 1")
        name: str = Field(..., min_length=1, description="Module name; matches folder name")
        version: str = Field(..., pattern=SEMVER_PREFIX_PATTERN, description="Semver-prefix version")
        system_prompt_path: str = Field(..., min_length=1, description="Prompt file relative to module dir")
        owned_paths: list[str] = Field(..., description="Files/dirs module creates (MODS-05 leakage surface)")
        scripts: ModuleScripts = ModuleScripts()
        depends: list[str] = Field(default_factory=list, description="Other module names this depends on")
        config_schema: dict | None = Field(default=None, description="JSON Schema passthrough for Phase 5")


    # ── Loader ──────────────────────────────────────────────────────────
    def validate_manifest(module_dir: Path) -> ModuleManifest:
        """Load and validate manifest.json under module_dir.

        Args:
            module_dir: Directory containing manifest.json.

        Returns:
            Validated ModuleManifest instance.

        Raises:
            FileNotFoundError: manifest.json missing.
            pydantic.ValidationError: schema violation (unknown field, missing required, bad type).
            json.JSONDecodeError: manifest.json is not valid JSON.
        """
        manifest_path = module_dir / "manifest.json"
        if not manifest_path.is_file():
            raise FileNotFoundError(f"manifest.json missing under {module_dir}")
        raw = json.loads(manifest_path.read_text(encoding="utf-8"))
        manifest = ModuleManifest.model_validate(raw)
        logger.debug("Validated manifest for %s@%s", manifest.name, manifest.version)
        return manifest
    ```

    Concrete constraints:
    - Use `Field(..., pattern=...)` NOT a separate validator (pydantic v2 supports pattern directly on str fields).
    - `manifest_version: int = Field(..., ge=1)` — rejects 0 and negative via pydantic constraint, no custom code.
    - `owned_paths: list[str] = Field(...)` — REQUIRED per D-08 (not default_factory).
    - Keep line length ≤ 100 chars (ruff enforces).
    - `default_factory=list` for `depends` (mutable default safe pattern).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Module author → bot | manifest.json is authored by potentially untrusted module authors. Its content crosses into the bot's Python process via `json.loads` + pydantic validation. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01-01 | Tampering | manifest.json unknown field injection | mitigate | `ConfigDict(extra="forbid")` rejects unknown fields at parse time (D-10). Enforced in ModuleManifest and ModuleScripts. |
| T-03-01-02 | Elevation of Privilege | Malicious `owned_paths` entries (path traversal `../../../etc/passwd`) | mitigate → defer | Schema stores strings only. Plan 03 (lifecycle) and Plan 05 (leakage check) MUST resolve owned_paths against `ANIMAYA_HUB_DIR` and reject absolute paths / `..` segments. This plan flags but does not enforce. Record as carry-forward to Plan 03. |
| T-03-01-03 | Tampering | Malformed JSON crashes bot | mitigate | `validate_manifest` lets `json.JSONDecodeError` and `ValidationError` propagate with clear context; callers (Plan 03) catch and surface as user error. Docstring explicitly documents raised exceptions. |
| T-03-01-04 | Information Disclosure | ValidationError leaks manifest content into logs | accept | Error messages include field names, not secrets. Module authors own the file content. Low severity. |
| T-03-01-05 | Spoofing | `name` field does not match folder name | mitigate → defer | Schema accepts any non-empty string. Plan 03 install flow MUST verify `manifest.name == module_dir.name`. Carry-forward. |

    ```python
    """CLAUDE.md assembler (Phase 3, MODS-04).

    Rebuilds ``CLAUDE.md`` from the core base template + installed-module prompts
    at every bot startup and at the end of every install/uninstall (D-18).

    Order: base template → per-module sections sorted by ``installed_at`` (D-16),
    each wrapped in ``<module name="...">...</module>`` (D-17).
    """
    from __future__ import annotations

    import logging
    import os
    from pathlib import Path

    from bot.modules.manifest import validate_manifest
    from bot.modules.registry import read_registry

    logger = logging.getLogger(__name__)

    # Repo root: three levels up from this file (bot/modules/assembler.py)
    _REPO_ROOT = Path(__file__).resolve().parent.parent.parent
    DEFAULT_REPO_ROOT: Path = _REPO_ROOT
    DEFAULT_MODULES_ROOT: Path = _REPO_ROOT / "modules"
    DEFAULT_TEMPLATE_PATH: Path = _REPO_ROOT / "bot" / "templates" / "CLAUDE.md"

    # Markers preserved from Phase 1 stub for test compatibility
    MARKER_START = "<!-- module-prompts-start -->"
    MARKER_END = "<!-- module-prompts-end -->"
    EMPTY_MODULES_COMMENT = "<!-- No modules installed -->"


    def _escape_module_content(text: str) -> str:
        """Prevent a module's prompt from closing another module's XML tag.

        Replaces ``</module>`` inside prompt content with an HTML-entity-escaped
        variant. Opening tags are left alone; only the terminator is dangerous.
        """
        return text.replace("</module>", "&lt;/module&gt;")


    def _render_module_section(name: str, prompt_text: str) -> str:
        """Wrap a module prompt in its XML section (D-17)."""
        safe = _escape_module_content(prompt_text.strip())
        return f'<module name="{name}">\n{safe}\n</module>'


    def _atomic_write(path: Path, content: str) -> None:
        path.parent.mkdir(parents=True, exist_ok=True)
        tmp = path.with_suffix(path.suffix + ".tmp")
        tmp.write_text(content, encoding="utf-8")
        os.replace(tmp, path)


    def assemble_claude_md(
        hub_dir: Path,
        *,
        modules_root: Path | None = None,
        repo_root: Path | None = None,
        output_path: Path | None = None,
        template_path: Path | None = None,
    ) -> Path:
        """Rebuild CLAUDE.md; return path written.

        Args:
            hub_dir: Directory containing registry.json.
            modules_root: Directory holding module folders. Default: ``<repo>/modules``.
            repo_root: Project root (used to locate default output path).
            output_path: Where to write CLAUDE.md. Default: ``hub_dir / "CLAUDE.md"``
                (tests use this). Production (bot/main.py) may pass the repo-root
                CLAUDE.md path per D-19.
            template_path: Base template. Default: ``bot/templates/CLAUDE.md``.
        """
        modules_root = (modules_root or DEFAULT_MODULES_ROOT).resolve()
        template_path = (template_path or DEFAULT_TEMPLATE_PATH).resolve()
        output_path = (output_path or (hub_dir / "CLAUDE.md")).resolve()

        # 1. Base template (D-19)
        if template_path.is_file():
            base = template_path.read_text(encoding="utf-8").rstrip() + "\n"
        else:
            logger.warning("base template missing at %s — using empty", template_path)
            base = ""

        # 2. Registry read (D-18 defensive: missing file → empty registry)
        registry = read_registry(hub_dir)
        entries = sorted(registry["modules"], key=lambda e: e.get("installed_at", ""))

        # 3. Per-module sections (D-16 install order, D-17 XML wrap)
        sections: list[str] = []
        for entry in entries:
            name = entry["name"]
            module_dir = modules_root / name
            if not module_dir.is_dir():
                logger.warning("module dir missing for %s at %s", name, module_dir)
                sections.append(f'<module name="{name}">\n<!-- module dir not found -->\n</module>')
                continue
            try:
                manifest = validate_manifest(module_dir)
            except Exception as exc:  # noqa: BLE001 — defensive assembler must not fail boot
                logger.warning("manifest invalid for %s: %s", name, exc)
                sections.append(f'<module name="{name}">\n<!-- manifest invalid -->\n</module>')
                continue
            prompt_file = module_dir / manifest.system_prompt_path
            if not prompt_file.is_file():
                logger.warning("prompt missing for %s at %s", name, prompt_file)
                sections.append(f'<module name="{name}">\n<!-- prompt not found -->\n</module>')
                continue
            sections.append(_render_module_section(name, prompt_file.read_text(encoding="utf-8")))

        # 4. Assemble body between markers
        if sections:
            body = "\n\n".join(sections)
        else:
            body = EMPTY_MODULES_COMMENT

        output = (
            f"{base}\n"
            f"{MARKER_START}\n"
            f"{body}\n"
            f"{MARKER_END}\n"
        )
        _atomic_write(output_path, output)
        logger.info("CLAUDE.md assembled at %s (%d modules)", output_path, len(sections))
        return output_path
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Module author → CLAUDE.md → Claude | Installed modules' prompt.md content flows into Claude's system prompt verbatim. Malicious content could prompt-inject another module or the core bot. |
| Filesystem → assembler | Assembler reads registry + per-module prompts; a compromised prompt file affects Claude's behavior. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-04-01 | Tampering | Prompt injection: malicious module's prompt.md contains `</module>` to escape its section and inject instructions into a later module's section | mitigate | `_escape_module_content` replaces `</module>` with `&lt;/module&gt;` so the XML wrapper is inviolable. Test covers at roundtrip stage (Plan 05). |
| T-03-04-02 | Spoofing | Module prompt impersonates core-bot instructions (`You are now in developer mode ignore previous rules`) | accept | XML wrapper (`<module name="X">`) makes the source clear to Claude. Defence-in-depth by core base-template language. Cannot prevent via purely-structural means; documented as module-author trust boundary. |
| T-03-04-03 | Tampering | assembler.py crash aborts bot boot (D-18 violation) | mitigate | Broad `except Exception` on per-module load with warning log; assembler always writes a valid CLAUDE.md even on partial failure. In lifecycle.py, reassembly failures are caught and logged, not raised. |
| T-03-04-04 | Tampering | Partial-write corruption of CLAUDE.md (crash mid-write) | mitigate | `_atomic_write` uses temp + `os.replace`. Consistent with registry write pattern. |
| T-03-04-05 | Denial of Service | Enormous prompt.md (1GB) consumes memory | accept | Modules are trusted; no size cap in Phase 3. Phase 5 dashboard install path may add one. Carry-forward. |
| T-03-04-06 | Information Disclosure | prompt.md contains secrets that get written to CLAUDE.md (world-readable in LXC) | accept | Module authors own their prompt content; project convention is "no secrets in prompts, use env vars". Documented in module authoring guide (Plan 05). |
</threat_model>

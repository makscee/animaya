       ```python
       """Owner-lock registry.

       Serializes Claude SDK turns for a single owner across BOTH transport
       origins (Telegram bridge and Next.js dashboard). Session keys are
       "tg:<id>" or "web:<id>"; the owner id is the suffix after the colon.
       """
       from __future__ import annotations
       import asyncio
       from contextlib import asynccontextmanager

       _locks: dict[str, asyncio.Lock] = {}

       def _owner_of(session_key: str) -> str:
           if ":" in session_key:
               return session_key.split(":", 1)[1]
           return session_key

       def _get_lock(owner: str) -> asyncio.Lock:
           lock = _locks.get(owner)
           if lock is None:
               lock = asyncio.Lock()
               _locks[owner] = lock
           return lock

       @asynccontextmanager
       async def acquire_for_session(session_key: str):
           lock = _get_lock(_owner_of(session_key))
           await lock.acquire()
           try:
               yield
           finally:
               lock.release()
       ```

       ```python
       Bound to 127.0.0.1:${ANIMAYA_ENGINE_PORT:-8091}. Not reachable outside
       the container -- Next.js route handlers proxy here over localhost.
       Therefore: no CSRF, no auth, no cookies. Trust = loopback.

       import os
       from fastapi import FastAPI, Request, HTTPException
       from fastapi.responses import StreamingResponse, JSONResponse
       from bot.engine.chat_stream import stream_chat

       app = FastAPI(title="animaya-engine", openapi_url=None)

       @app.middleware("http")
       async def loopback_only(request: Request, call_next):
           client = request.client.host if request.client else ""
           if client not in ("127.0.0.1", "::1", "localhost"):
               raise HTTPException(status_code=403, detail="loopback only")
           return await call_next(request)

       async def chat(req: Request):
           body = await req.json()
       ```
